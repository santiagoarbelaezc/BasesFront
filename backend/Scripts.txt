CREATE SEQUENCE CURSO_SEQ
  START WITH 1
  INCREMENT BY 1
  NOCACHE
  NOCYCLE;

CREATE OR REPLACE PROCEDURE INSERTAR_CURSO (
  P_NOMBRE IN CURSO.NOMBRE%TYPE,
  P_DESCRIPCION IN CURSO.DESCRIPCION%TYPE
)
AS
BEGIN
  INSERT INTO CURSO (CURSO_ID, NOMBRE, DESCRIPCION)
  VALUES (CURSO_SEQ.NEXTVAL, P_NOMBRE, P_DESCRIPCION);
END;
/
CREATE OR REPLACE PROCEDURE OBTENER_CURSOS (
  CURSOR_OUT OUT SYS_REFCURSOR
)
AS
BEGIN
  OPEN CURSOR_OUT FOR
    SELECT CURSO_ID, NOMBRE, DESCRIPCION
    FROM CURSO
    ORDER BY CURSO_ID;
END;
/
CREATE OR REPLACE PROCEDURE ACTUALIZAR_CURSO (
  P_CURSO_ID IN CURSO.CURSO_ID%TYPE,
  P_NOMBRE IN CURSO.NOMBRE%TYPE,
  P_DESCRIPCION IN CURSO.DESCRIPCION%TYPE
)
AS
BEGIN
  UPDATE CURSO
  SET NOMBRE = P_NOMBRE,
      DESCRIPCION = P_DESCRIPCION
  WHERE CURSO_ID = P_CURSO_ID;
END;
/
CREATE OR REPLACE PROCEDURE ELIMINAR_CURSO (
  P_CURSO_ID IN CURSO.CURSO_ID%TYPE
)
AS
BEGIN
  DELETE FROM CURSO WHERE CURSO_ID = P_CURSO_ID;
END;
/
